def dfs(graph, start, goal):
    stack = [(start, [start])]
    visited = set()

    while stack:
        node, path = stack.pop()

        if node == goal:
            return path

        if node not in visited:
            visited.add(node)

            # Add neighbors in reverse lexicographic order
            for neighbor in reversed(graph[node]):
                if neighbor not in visited:
                    stack.append((neighbor, path + [neighbor]))

    return None


# Example usage
graph = {
    0: [1, 2],
    1: [3, 4],
    2: [5],
    3: [],
    4: [],
    5: []
}

start = 0
goal = 5

result = dfs(graph, start, goal)

print("DFS Path:", result)


BFS (Breadth First Search):

from collections import deque

def bfs(graph, start, goal):
    queue = deque([(start, [start])])
    visited = set()

    while queue:
        node, path = queue.popleft()

        if node == goal:
            return path

        if node not in visited:
            visited.add(node)

            for neighbor in graph[node]:
                if neighbor not in visited:
                    queue.append((neighbor, path + [neighbor]))

    return None


# Example usage
graph = {
    0: [1, 2],
    1: [3, 4],
    2: [5],
    3: [],
    4: [],
    5: []
}

start = 0
goal = 5

result = bfs(graph, start, goal)

print("BFS Path:", result)


Lowest Cost First Search:

import heapq

def lowest_cost_first_search(graph, start, goal):
    priority_queue = [(0, start, [start])]
    visited = set()

    while priority_queue:
        cost, node, path = heapq.heappop(priority_queue)

        if node == goal:
            return path, cost

        if node not in visited:
            visited.add(node)

            for neighbor, weight in graph[node]:
                if neighbor not in visited:
                    heapq.heappush(
                        priority_queue,
                        (cost + weight, neighbor, path + [neighbor])
                    )

    return None, float('inf')


# Example weighted graph
graph = {
    'A': [('B', 1), ('C', 4)],
    'B': [('D', 2), ('E', 5)],
    'C': [('F', 1)],
    'D': [],
    'E': [],
    'F': []
}

start = 'A'
goal = 'F'

path, cost = lowest_cost_first_search(graph, start, goal)

print("Lowest Cost Path:", path)
print("Total Cost:", cost)


Explanation

DFS, BFS, and Lowest Cost First Search are important search algorithms used in Artificial Intelligence for problem solving and path finding. DFS (Depth First Search) explores nodes deeply along one branch before backtracking, making it useful when the solution may be far from the root. BFS (Breadth First Search) explores all neighboring nodes level by level and is useful for finding the shortest path in an unweighted graph. Lowest Cost First Search selects the path with the minimum cumulative cost and is mainly used in weighted graphs where different paths have different costs. In this experiment, we implemented these search algorithms using Python and observed how each algorithm finds the path from the start node to the goal node. This practical helped in understanding graph traversal techniques and the working of uninformed search strategies in AI.
